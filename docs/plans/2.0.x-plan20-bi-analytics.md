## Contexto

Histrix tiene ~169 pantallas de estadisticas/reportes repartidas en todos los
modulos. Son queries SQL hardcodeadas en XMLs que renderizan tablas HTML. No
hay graficos, no hay export, no hay filtros dinamicos.

SDA necesita un sistema de analytics que:
- Exponga KPIs por dominio via API
- Permita filtrar por periodo, entidad, centro de costo, etc.
- Tenga graficos en el frontend (evolucion, comparativo, ranking)
- Exporte a CSV/Excel
- Sea extensible (agregar un reporte nuevo = agregar un query + endpoint)

## Principios

1. **Reporting es read-only.** Cero writes, cero mutations, cero side effects.
2. **Las queries estan en sqlc.** Type-safe, auditables, performantes.
3. **Cada endpoint devuelve data + metadata.** El frontend decide como graficar.
4. **Export es un format parameter, no un endpoint separado.** `?format=csv`.

```go
// Column define una columna exportable.
type Column struct {
    Header string
    Key    string
    Format string  // "text", "number", "currency", "date", "percent"
}

// Row es un map generico de datos.
type Row map[string]any

// WriteCSV escribe rows como CSV al writer.
func WriteCSV(w io.Writer, columns []Column, rows []Row) error {
    cw := csv.NewWriter(w)
    defer cw.Flush()

    // Header
    headers := make([]string, len(columns))
    for i, c := range columns { headers[i] = c.Header }
    cw.Write(headers)

    // Data
    for _, row := range rows {
        record := make([]string, len(columns))
        for i, c := range columns {
            record[i] = formatValue(row[c.Key], c.Format)
        }
        cw.Write(record)
    }
    return nil
}

// WriteExcel escribe rows como XLSX.
func WriteExcel(w io.Writer, sheetName string, columns []Column, rows []Row) error {
    f := excelize.NewFile()
    // ... header row with bold style
    // ... data rows with number formatting
    // ... auto-width columns
    return f.Write(w)
}
```

### Handler base pattern

```go
// analytics.go — pattern comun para todos los reportes
// Threshold: rows > 10_000 en excel → async path (pending export + WS event)
const asyncExportThreshold = 10_000

func (h *Analytics) handleReport(w http.ResponseWriter, r *http.Request,
    exportName string, columns []export.Column, fetchFn func() ([]export.Row, error)) {

    rows, err := fetchFn()
    if err != nil { ... }

    format := r.URL.Query().Get("format")
    switch format {
    case "csv":
        // CSV es cheap — siempre sincrono
        w.Header().Set("Content-Type", "text/csv")
        w.Header().Set("Content-Disposition", "attachment; filename=report.csv")
        export.WriteCSV(w, columns, rows)
    case "excel":
        if len(rows) > asyncExportThreshold {
            // Async: crear pending export, devolver 202 con ID
            pe := h.svc.CreatePendingExport(ctx, slug, userID, exportName, columns, rows)
            w.WriteHeader(http.StatusAccepted)
            json.NewEncoder(w).Encode(map[string]any{
                "export_id": pe.ID,
                "status":    "pending",
                "message":   "El archivo se esta generando. Recibiras un evento WS cuando este listo.",
            })
            return
        }
        // Sync si cabe
        w.Header().Set("Content-Type", "application/vnd.openxmlformats-officedocument.spreadsheetml.sheet")
        w.Header().Set("Content-Disposition", "attachment; filename=report.xlsx")
        export.WriteExcel(w, "Reporte", columns, rows)
    default:
        // JSON (default)
        json.NewEncoder(w).Encode(map[string]any{
            "columns": columns,
            "rows":    rows,
            "meta":    map[string]any{"count": len(rows)},
        })
    }
}
```

### Worker + WebSocket event

```go
// Worker pool (spawned en main, tamaño = runtime.NumCPU())
// Consume erp_pending_exports status='pending', procesa, sube a MinIO.
//
// El worker NO guarda filas en DB — guarda los PARAMS del reporte
// (date_from, date_to, entity_id, etc.) + columns_def, y al correr
// re-ejecuta la query con esos params. Esto evita blobs enormes en DB
// y da data fresca si los datos cambiaron entre request y run.
func (s *Analytics) processExportJob(ctx context.Context, job *PendingExport) {
    if err := s.repo.MarkExportRunning(ctx, job.ID); err != nil {
        slog.Error("mark running failed", "job", job.ID, "err", err)
        return
    }

    // Despachar a la fetch function correcta segun ExportName.
    // Cada reporte registrado conoce sus params (typed via JSONB decode).
    fetchFn, ok := s.exportFetchers[job.ExportName]
    if !ok {
        s.failJob(ctx, job, fmt.Errorf("unknown export name: %s", job.ExportName))
        return
    }

    rows, err := fetchFn(ctx, job.TenantID, job.Params)
    if err != nil {
        s.failJob(ctx, job, fmt.Errorf("fetch rows: %w", err))
        return
    }

    var buf bytes.Buffer
    if err := export.WriteExcel(&buf, "Reporte", job.ColumnsDef, rows); err != nil {
        s.failJob(ctx, job, fmt.Errorf("write excel: %w", err))
        return
    }

    key := fmt.Sprintf("exports/%s/%s.xlsx", job.TenantID, job.ID)
    if err := s.storage.Put(ctx, key, &buf, storage.PutOptions{
        ContentType: "application/vnd.openxmlformats-officedocument.spreadsheetml.sheet",
    }); err != nil {
        s.failJob(ctx, job, fmt.Errorf("upload: %w", err))
        return
    }

    s.repo.MarkExportReady(ctx, job.ID, key)
    s.publishExportEvent(job, "ready", key)
}

func (s *Analytics) failJob(ctx context.Context, job *PendingExport, err error) {
    s.repo.MarkExportFailed(ctx, job.ID, err.Error())
    s.publishExportEvent(job, "failed", "")
    slog.Error("export job failed", "job", job.ID, "export", job.ExportName, "err", err)
}

func (s *Analytics) publishExportEvent(job *PendingExport, status, key string) {
    s.nats.Publish("tenant."+job.TenantID+".notify.export_ready", map[string]any{
        "export_id":   job.ID,
        "user_id":     job.UserID,
        "export_name": job.ExportName,
        "status":      status,
        "download_url": fmt.Sprintf("/v1/erp/exports/%s/download", job.ID),
    })
}
```

### Stale job sanity check (cleanup)

Un worker que crashea a mitad deja el row en `status='running'` con `started_at` setead.
Para evitar que queden huerfanos, el worker pool ejecuta un **sanity sweep cada 5 minutos**
al inicio de cada tick:

```go
// Busca jobs 'running' que empezaron hace >15 minutos (max reasonable runtime).
// Los marca como 'failed' y emite evento WS al usuario.
func (s *Analytics) sweepStaleJobs(ctx context.Context) {
    stale, _ := s.repo.ListStaleRunningExports(ctx, 15*time.Minute)
    for _, job := range stale {
        s.repo.MarkExportFailed(ctx, job.ID, "worker crashed or timed out")
        s.publishExportEvent(job, "failed", "")
        slog.Warn("swept stale export", "job", job.ID, "started_at", job.StartedAt)
    }
}
```

Y un sweep mas lento (cada 1 hora) que borra rows con `expires_at < now()` + el blob
de MinIO asociado. Ambos sweeps corren en el main process, no requieren cron externo.

### Download endpoint (sin presigned URL)

```go
// GET /v1/erp/exports/{id}/download
// Sirve el blob directamente desde MinIO con access control del handler.
// No usamos presigned URLs — pkg/storage no las expone y el handler maneja
// access control natural via JWT + tenant + owner check.
func (h *Analytics) DownloadExport(w http.ResponseWriter, r *http.Request) {
    id := chi.URLParam(r, "id")
    pe := h.svc.GetPendingExport(ctx, slug, id)
    if pe.UserID != userID { http.Error(w, "forbidden", 403); return }
    if pe.Status != "ready" { http.Error(w, "not ready", 409); return }

    reader, _ := h.storage.Get(ctx, pe.FileKey)
    defer reader.Close()
    w.Header().Set("Content-Type", "application/vnd.openxmlformats-officedocument.spreadsheetml.sheet")
    w.Header().Set("Content-Disposition", "attachment; filename="+pe.ExportName+".xlsx")
    io.Copy(w, reader)
}
```

### Queries de ejemplo

```sql
-- name: AnalyticsIncomeExpense :many
-- Resultado mensual: ingresos vs egresos
SELECT
    to_char(je.date, 'YYYY-MM') AS period,
    SUM(CASE WHEN a.account_type = 'income' THEN jl.credit - jl.debit ELSE 0 END) AS income,
    SUM(CASE WHEN a.account_type = 'expense' THEN jl.debit - jl.credit ELSE 0 END) AS expense
FROM erp_journal_entries je
JOIN erp_journal_lines jl ON jl.entry_id = je.id AND jl.tenant_id = je.tenant_id
JOIN erp_accounts a ON a.id = jl.account_id
WHERE je.tenant_id = @tenant_id
    AND je.status = 'posted'
    AND je.date BETWEEN @date_from AND @date_to
GROUP BY to_char(je.date, 'YYYY-MM')
ORDER BY period;

-- name: AnalyticsCashFlow :many
-- Flujo de caja mensual
SELECT
    to_char(date, 'YYYY-MM') AS period,
    SUM(CASE WHEN movement_type IN ('cash_in', 'bank_deposit', 'check_received') THEN amount ELSE 0 END) AS inflow,
    SUM(CASE WHEN movement_type IN ('cash_out', 'bank_withdrawal', 'check_issued') THEN amount ELSE 0 END) AS outflow
FROM erp_treasury_movements
WHERE tenant_id = @tenant_id
    AND status = 'confirmed'
    AND date BETWEEN @date_from AND @date_to
GROUP BY to_char(date, 'YYYY-MM')
ORDER BY period;

-- name: AnalyticsAging :many
-- Aging de cuentas por cobrar
SELECT
    e.id AS entity_id, e.name AS entity_name,
    SUM(CASE WHEN am.balance > 0 AND i.due_date >= CURRENT_DATE THEN am.balance ELSE 0 END) AS current,
    SUM(CASE WHEN am.balance > 0 AND i.due_date BETWEEN CURRENT_DATE - 30 AND CURRENT_DATE - 1 THEN am.balance ELSE 0 END) AS days_1_30,
    SUM(CASE WHEN am.balance > 0 AND i.due_date BETWEEN CURRENT_DATE - 60 AND CURRENT_DATE - 31 THEN am.balance ELSE 0 END) AS days_31_60,
    SUM(CASE WHEN am.balance > 0 AND i.due_date BETWEEN CURRENT_DATE - 90 AND CURRENT_DATE - 61 THEN am.balance ELSE 0 END) AS days_61_90,
    SUM(CASE WHEN am.balance > 0 AND i.due_date < CURRENT_DATE - 90 THEN am.balance ELSE 0 END) AS days_over_90
FROM erp_account_movements am
JOIN erp_entities e ON e.id = am.entity_id
LEFT JOIN erp_invoices i ON i.id = am.invoice_id
WHERE am.tenant_id = @tenant_id
    AND am.direction = @direction  -- 'receivable' o 'payable'
    AND am.balance > 0
GROUP BY e.id, e.name
ORDER BY SUM(am.balance) DESC;
```

### Queries de ejemplo

```sql
-- name: AnalyticsStockRotation :many
-- Rotacion de stock (ultimos N meses)
-- Rotation = consumed / average_stock (not current_stock, which is a point-in-time snapshot).
-- Average stock approximated as (current_stock + consumed_qty) / 2 (beginning ≈ current + consumed).
SELECT
    a.id, a.code, a.name,
    COALESCE(sl.quantity, 0) AS current_stock,
    COALESCE(a.last_cost, 0) AS unit_cost,
    COALESCE(sl.quantity, 0) * COALESCE(a.last_cost, 0) AS stock_value,
    COALESCE(consumed.qty, 0) AS consumed_qty,
    CASE WHEN (COALESCE(sl.quantity, 0) + COALESCE(consumed.qty, 0)) > 0
        THEN COALESCE(consumed.qty, 0) / ((COALESCE(sl.quantity, 0) + COALESCE(consumed.qty, 0)) / 2.0)
        ELSE 0
    END AS rotation_index
FROM erp_articles a
LEFT JOIN erp_stock_levels sl ON sl.article_id = a.id AND sl.tenant_id = a.tenant_id
LEFT JOIN (
    SELECT article_id, SUM(ABS(quantity)) AS qty
    FROM erp_stock_movements
    WHERE tenant_id = @tenant_id AND movement_type = 'out'
        AND created_at >= @date_from
    GROUP BY article_id
) consumed ON consumed.article_id = a.id
WHERE a.tenant_id = @tenant_id AND a.active = true
ORDER BY rotation_index DESC;

-- name: AnalyticsProductionEfficiency :many
-- NOTE: end_date and start_date are DATE type. DATE - DATE in PostgreSQL
-- returns INTEGER (number of days) directly — no EXTRACT/EPOCH needed.
SELECT
    pc.name AS center_name,
    COUNT(*) AS total_orders,
    COUNT(*) FILTER (WHERE po.status = 'completed') AS completed,
    AVG(po.end_date - po.start_date)
        FILTER (WHERE po.status = 'completed' AND po.end_date IS NOT NULL AND po.start_date IS NOT NULL)
        AS avg_days
FROM erp_production_orders po
LEFT JOIN erp_production_centers pc ON pc.id = po.center_id
WHERE po.tenant_id = @tenant_id
    AND po.date BETWEEN @date_from AND @date_to
GROUP BY pc.name
ORDER BY total_orders DESC;
```

Este endpoint ejecuta ~12 queries simples (COUNT/SUM) en paralelo usando
goroutines. Cada query es rapida por si sola; paralelizarlas hace que el
endpoint responda en <200ms.

**Pool exhaustion protection (obligatorio):** Con N usuarios concurrentes, 12
queries paralelas multiplican el uso de conexiones. Usar un `semaphore` a nivel
del handler que limita 4 queries concurrentes por request — las restantes esperan.
Esto es **codigo real en fase 3**, no recomendacion:

```go
import "golang.org/x/sync/semaphore"

var dashboardSem = semaphore.NewWeighted(4) // max 4 queries paralelas por request

func (h *Analytics) DashboardKPIs(w http.ResponseWriter, r *http.Request) {
    ctx := r.Context()
    var kpis DashboardKPIs
    var mu sync.Mutex
    var wg sync.WaitGroup
    errs := make(chan error, 12)  // buffer = cant de queries

    runQuery := func(name string, fn func(context.Context) error) {
        wg.Add(1)
        go func() {
            defer wg.Done()
            if err := dashboardSem.Acquire(ctx, 1); err != nil {
                // ctx cancelled o deadline — reportar y abort, no silenciar
                errs <- fmt.Errorf("acquire sem for %s: %w", name, err)
                return
            }
            defer dashboardSem.Release(1)
            if err := fn(ctx); err != nil {
                slog.Error("dashboard query failed", "query", name, "err", err)
                errs <- fmt.Errorf("%s: %w", name, err)
            }
        }()
    }

    runQuery("month_revenue", func(ctx context.Context) error {
        v, err := h.svc.GetMonthRevenue(ctx, slug)
        if err == nil { mu.Lock(); kpis.MonthRevenue = v; mu.Unlock() }
        return err
    })
    // ... mismas 11 queries mas

    wg.Wait()
    close(errs)

    // Si TODAS las queries fallaron → 500. Si solo algunas → 206 Partial Content
    // con los KPIs que si pudieron calcularse + lista de errores en el body.
    var errList []string
    for err := range errs { errList = append(errList, err.Error()) }
    if len(errList) == 12 {
        http.Error(w, "all dashboard queries failed", 500)
        return
    }
    status := http.StatusOK
    if len(errList) > 0 { status = http.StatusPartialContent }
    w.WriteHeader(status)
    json.NewEncoder(w).Encode(map[string]any{
        "kpis":   kpis,
        "errors": errList,  // frontend muestra banner "algunos datos no pudieron cargarse"
    })
}
```

## Consideraciones de performance

- Todas las queries de analytics usan los indices ya existentes de plan 17.
  Si al desarrollar una query el EXPLAIN muestra Seq Scan, crear el indice en
  esta migracion 039 (no en la migration del dominio — aislar cambios de plan 20).
- Para datasets grandes (>100k rows), el endpoint devuelve datos paginados
  y el export genera el archivo en background con descarga async.
- **Materialized views NO en V1.** Si una query degrada >200ms en produccion
  con volumen real, escalar a V1.1 con materialized view + refresh CONCURRENTLY
  por cron nocturno. **Decision gate:** al detectar la primera query lenta en
  prod, planear V1.1 — no antes.
