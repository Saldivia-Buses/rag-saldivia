### Fase 5: JWT access + refresh token rotation

**Archivos:**
- `apps/web/src/lib/auth/jwt.ts`
- `apps/web/src/app/api/auth/login/route.ts`
- `apps/web/src/app/api/auth/refresh/route.ts`
- `apps/web/src/app/api/auth/logout/route.ts`

**Gap mas grande vs la industria:** token unico de 24h. Todos usan access corto + refresh largo.

**Nuevo flujo:**

```
Login:
  → access token (15m, cookie auth_token, SameSite=Lax)
  → refresh token (7d, cookie refresh_token, HttpOnly, SameSite=Strict, Path=/api/auth/refresh)

Cada request:
  → proxy.ts lee auth_token (sin cambios — misma cookie, mismo JWT)

Refresh (antes de que expire el access):
  → POST /api/auth/refresh con refresh_token cookie
  → verificar refresh token valido + no revocado
  → revocar refresh viejo en Redis: SET revoked:{jti} 1 EX {ttl_restante}
  → emitir nuevo access + nuevo refresh
  → setear ambas cookies

Logout:
  → revocar access jti + refresh jti en Redis
  → clear ambas cookies
```

**En jwt.ts — funciones nuevas:**

```typescript
import { ACCESS_TOKEN_EXPIRY, REFRESH_TOKEN_EXPIRY, REFRESH_TOKEN_MAX_AGE_S } from "@rag-saldivia/config"

export async function createAccessToken(claims: Omit<JwtClaims, "iat" | "exp" | "jti">): Promise<string> {
  return new SignJWT({ ...claims, type: "access" })
    .setProtectedHeader({ alg: "HS256" })
    .setIssuedAt()
    .setExpirationTime(ACCESS_TOKEN_EXPIRY)
    .setJti(crypto.randomUUID())
    .sign(getSecret())
}

export async function createRefreshToken(sub: string): Promise<string> {
  return new SignJWT({ sub, type: "refresh" })
    .setProtectedHeader({ alg: "HS256" })
    .setIssuedAt()
    .setExpirationTime(REFRESH_TOKEN_EXPIRY)
    .setJti(crypto.randomUUID())
    .sign(getSecret())
}

export function makeRefreshCookie(token: string): string {
  const isProduction = process.env["NODE_ENV"] === "production"
  return [
    `refresh_token=${encodeURIComponent(token)}`,
    `Max-Age=${REFRESH_TOKEN_MAX_AGE_S}`,
    "Path=/api/auth/refresh",
    "HttpOnly",
    "SameSite=Strict",
    ...(isProduction ? ["Secure"] : []),
  ].join("; ")
}

export function makeClearRefreshCookie(): string {
  return "refresh_token=; Max-Age=0; Path=/api/auth/refresh; HttpOnly; SameSite=Strict"
}

export async function revokeToken(jti: string, expSeconds: number): Promise<void> {
  const ttl = expSeconds - Math.floor(Date.now() / 1000)
  if (ttl > 0) {
    await getRedisClient().set(`revoked:${jti}`, "1", "EX", ttl).catch(() => {})
  }
}
```

**Mantener backward compat:**
- `createJwt()` sigue existiendo (alias de `createAccessToken`) — no romper imports
- `makeAuthCookie()` ahora usa `ACCESS_TOKEN_EXPIRY` en vez de `getExpiry()`
- `proxy.ts` sin cambios — sigue leyendo `auth_token` cookie

**En refresh/route.ts — revocar token viejo:**

```typescript
export async function POST(request: Request) {
  // Extraer refresh token de cookie
  const refreshToken = extractRefreshTokenFromCookie(request)
  if (!refreshToken) return apiError("No refresh token", 401)

  const claims = await verifyJwt(refreshToken)
  if (!claims || claims.type !== "refresh") return apiError("Invalid refresh token", 401)
  if (claims.jti && await isRevoked(claims.jti)) return apiError("Token revoked", 401)

  // Revocar refresh viejo
  if (claims.jti) await revokeToken(claims.jti, claims.exp)

  const user = await getUserById(Number(claims.sub))
  if (!user || !user.active) return apiError("Account disabled", 403)

  // Emitir nuevos tokens
  const newAccess = await createAccessToken({ sub: String(user.id), email: user.email, name: user.name, role: user.role })
  const newRefresh = await createRefreshToken(String(user.id))

  const response = NextResponse.json({ ok: true })
  response.headers.append("Set-Cookie", makeAuthCookie(newAccess))
  response.headers.append("Set-Cookie", makeRefreshCookie(newRefresh))
  return response
}
```

import { ADMIN_STATS_CACHE_TTL_S } from "@rag-saldivia/config"

const CACHE_KEY = "cache:admin:stats"

async function getAdminStats() {
  try {
    const cached = await getRedisClient().get(CACHE_KEY)
    if (cached) return JSON.parse(cached)
  } catch {}

  const [users, sessions, messages, roles, usersPresence] = await Promise.all([
    countUsers(), countSessions(), countMessages(), listRoles(), getUsersPresence(),
  ])
  const stats = { users, sessions, messages, roles, usersPresence }

  try {
    await getRedisClient().set(CACHE_KEY, JSON.stringify(stats), "EX", ADMIN_STATS_CACHE_TTL_S)
  } catch {}

  return stats
}
