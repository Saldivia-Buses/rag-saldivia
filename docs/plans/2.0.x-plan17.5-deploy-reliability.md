## Contexto

Despues de cambiar codigo y deployar, a veces queda codigo viejo running.
El deploy es no-determinístico: Docker reutiliza imagenes locales, binarios
no llevan version info, y no hay forma de verificar que lo que esta corriendo
es lo que se acaba de buildar.

### Raiz del problema

5 agujeros concretos:

| # | Problema | Impacto |
|---|----------|---------|
| 1 | `docker-compose.prod.yml` defaultea a `latest` | Docker reutiliza imagen local vieja |
| 2 | Dockerfiles no inyectan `-ldflags` | Binarios dicen `Version="dev"`, `GitSHA="unknown"` |
| 3 | `/v1/info` devuelve datos inutiles | Endpoint existe (pkg/server) pero sin ldflags siempre dice "dev" |
| 4 | `make versions` es un placeholder muerto | Llama a `sda versions` que no existe |
| 5 | `docker pull latest` no baja la nueva si ya tenes `latest` local | No-op silencioso |

### Que ya funciona (no tocar)

- `pkg/build/info.go` — completo, ldflags documentados, fallback a `debug.ReadBuildInfo()`
- `pkg/server/server.go` linea 117 — YA registra `r.Get("/v1/info", build.Handler(name))` para todos los servicios
- Health checks — todos los servicios tienen `/health` + healthcheck en compose
- Multi-stage builds — correcto (golang:1.25-alpine → distroless)
- `deploy/scripts/preflight.sh` — 13 checks de pre-deploy

#### 1b. Crear deploy script

**Archivo nuevo:** `deploy/scripts/deploy.sh`

Script que reemplaza el `docker compose up -d --build` manual:

```bash
#!/bin/bash
# deploy.sh — Deterministic deploy for SDA Framework
#
# Usage:
#   deploy/scripts/deploy.sh              # deploy all
#   deploy/scripts/deploy.sh auth chat    # deploy specific services
#
# What it does:
#   1. Reads VERSION files from each service
#   2. Sets GIT_SHA and BUILD_TIME
#   3. Generates .env.deploy with all version vars
#   4. Builds with --no-cache (optional flag: --cache for faster rebuilds)
#   5. docker compose up -d
#   6. Waits for health checks
#   7. Verifies /v1/info of each service matches expected SHA
#   8. Reports mismatch as error

set -euo pipefail

SCRIPT_DIR="$(cd "$(dirname "$0")" && pwd)"
ROOT_DIR="$(cd "$SCRIPT_DIR/../.." && pwd)"
DEPLOY_DIR="$ROOT_DIR/deploy"
COMPOSE_FILE="$DEPLOY_DIR/docker-compose.prod.yml"

# Git info
GIT_SHA=$(cd "$ROOT_DIR" && git rev-parse --short HEAD 2>/dev/null || echo "unknown")
BUILD_TIME=$(date -u +%Y-%m-%dT%H:%M:%SZ)

# Service → port mapping
declare -A SERVICE_PORTS=(
    [auth]=8001 [ws]=8002 [chat]=8003 [agent]=8004
    [notification]=8005 [platform]=8006 [ingest]=8007
    [feedback]=8008 [traces]=8009 [search]=8010
    [astro]=8011 [bigbrother]=8012 [erp]=8013
)

# Parse args
NO_CACHE="--no-cache"
SERVICES=()
for arg in "$@"; do
    if [ "$arg" = "--cache" ]; then
        NO_CACHE=""
    else
        SERVICES+=("$arg")
    fi
done

# If no services specified, deploy all Go services
if [ ${#SERVICES[@]} -eq 0 ]; then
    SERVICES=(auth ws chat agent search astro traces notification platform ingest feedback bigbrother erp)
fi

echo "═══ SDA Deploy ═══"
echo "  Git SHA:    $GIT_SHA"
echo "  Build time: $BUILD_TIME"
echo "  Services:   ${SERVICES[*]}"
echo ""

# Generate .env.deploy with version variables
ENV_DEPLOY="$DEPLOY_DIR/.env.deploy"
{
    echo "# Auto-generated by deploy.sh — $(date -u)"
    echo "GIT_SHA=$GIT_SHA"
    echo "BUILD_TIME=$BUILD_TIME"
    for svc in "${SERVICES[@]}"; do
        upper=$(echo "$svc" | tr '[:lower:]' '[:upper:]')
        version_file="$ROOT_DIR/services/$svc/VERSION"
        if [ -f "$version_file" ]; then
            version=$(cat "$version_file" | tr -d '[:space:]')
        else
            version="dev"
        fi
        echo "${upper}_VERSION=${version}"
    done
} > "$ENV_DEPLOY"

echo "  Generated $ENV_DEPLOY"

# Source .env.deploy so compose picks up the vars
set -a
source "$ENV_DEPLOY"
# Also source main .env if exists
[ -f "$DEPLOY_DIR/.env" ] && source "$DEPLOY_DIR/.env"
set +a

# Build
echo ""
echo "── Building ──"
docker compose -f "$COMPOSE_FILE" build $NO_CACHE "${SERVICES[@]}"

# Start
echo ""
echo "── Starting ──"
docker compose -f "$COMPOSE_FILE" up -d "${SERVICES[@]}"

# Wait for health
echo ""
echo "── Waiting for health checks (max 60s) ──"
TIMEOUT=60
for svc in "${SERVICES[@]}"; do
    port=${SERVICE_PORTS[$svc]:-0}
    if [ "$port" -eq 0 ]; then
        echo "  $svc: no port mapping, skipping health check"
        continue
    fi

    elapsed=0
    while [ $elapsed -lt $TIMEOUT ]; do
        code=$(curl -sf --max-time 2 -o /dev/null -w "%{http_code}" "http://localhost:$port/health" 2>/dev/null || echo "000")
        if [ "$code" = "200" ]; then
            break
        fi
        sleep 2
        elapsed=$((elapsed + 2))
    done

    if [ "$code" = "200" ]; then
        printf "  %-20s :%-5s \033[32mHEALTHY\033[0m\n" "$svc" "$port"
    else
        printf "  %-20s :%-5s \033[31mUNHEALTHY (code: %s)\033[0m\n" "$svc" "$port" "$code"
    fi
done

# Verify /v1/info matches expected SHA
echo ""
echo "── Verifying build info ──"
MISMATCHES=0
for svc in "${SERVICES[@]}"; do
    port=${SERVICE_PORTS[$svc]:-0}
    if [ "$port" -eq 0 ]; then continue; fi

    info=$(curl -sf --max-time 2 "http://localhost:$port/v1/info" 2>/dev/null || echo "{}")
    running_sha=$(echo "$info" | grep -o '"git_sha":"[^"]*"' | cut -d'"' -f4)

    if [ "$running_sha" = "$GIT_SHA" ]; then
        printf "  %-20s \033[32mOK\033[0m (sha: %s)\n" "$svc" "$running_sha"
    elif [ -z "$running_sha" ] || [ "$running_sha" = "unknown" ]; then
        printf "  %-20s \033[33mNO INFO\033[0m (endpoint returned: %s)\n" "$svc" "$running_sha"
        ((MISMATCHES++))
    else
        printf "  %-20s \033[31mMISMATCH\033[0m (expected: %s, got: %s)\n" "$svc" "$GIT_SHA" "$running_sha"
        ((MISMATCHES++))
    fi
done

echo ""
if [ $MISMATCHES -gt 0 ]; then
    echo "═══ DEPLOY COMPLETE WITH $MISMATCHES WARNING(S) ═══"
    echo "  Some services may be running stale code."
    echo "  Re-run with --no-cache or check Docker build cache."
    exit 1
else
    echo "═══ DEPLOY COMPLETE — ALL VERIFIED ═══"
fi
```

#### 2a. Implementar `sda versions`

**Archivo nuevo:** `tools/cli/cmd/versions.go`

```go
package cmd

import (
	"encoding/json"
	"fmt"
	"net/http"
	"os/exec"
	"strings"
	"time"

	"github.com/spf13/cobra"
)

var versionsCmd = &cobra.Command{
	Use:   "versions",
	Short: "Show running service versions vs current git SHA",
	Long: `Queries /v1/info on each running service and compares the git SHA
against the current HEAD. Shows a table with match/mismatch status.`,
	Run: runVersions,
}

type buildInfo struct {
	Service   string `json:"service"`
	Version   string `json:"version"`
	GitSHA    string `json:"git_sha"`
	BuildTime string `json:"build_time"`
	GoVersion string `json:"go_version"`
}

func runVersions(cmd *cobra.Command, args []string) {
	// Get current git SHA
	out, err := exec.Command("git", "rev-parse", "--short", "HEAD").Output()
	currentSHA := strings.TrimSpace(string(out))
	if err != nil || currentSHA == "" {
		currentSHA = "unknown"
	}

	fmt.Printf("Current HEAD: %s\n\n", currentSHA)
	fmt.Printf("%-20s %-10s %-10s %-22s %s\n", "SERVICE", "VERSION", "GIT SHA", "BUILD TIME", "STATUS")
	fmt.Println(strings.Repeat("─", 80))

	services := []struct {
		name string
		port int
	}{
		{"auth", 8001}, {"ws", 8002}, {"chat", 8003}, {"agent", 8004},
		{"notification", 8005}, {"platform", 8006}, {"ingest", 8007},
		{"feedback", 8008}, {"traces", 8009}, {"search", 8010},
		{"astro", 8011}, {"bigbrother", 8012}, {"erp", 8013},
	}

	client := &http.Client{Timeout: 2 * time.Second}

	for _, svc := range services {
		url := fmt.Sprintf("http://localhost:%d/v1/info", svc.port)
		resp, err := client.Get(url)
		if err != nil {
			fmt.Printf("%-20s %-10s %-10s %-22s \033[31mDOWN\033[0m\n",
				svc.name, "-", "-", "-")
			continue
		}
		defer resp.Body.Close()

		var info buildInfo
		if err := json.NewDecoder(resp.Body).Decode(&info); err != nil {
			fmt.Printf("%-20s %-10s %-10s %-22s \033[33mBAD RESPONSE\033[0m\n",
				svc.name, "-", "-", "-")
			continue
		}

		status := "\033[32mMATCH\033[0m"
		if info.GitSHA != currentSHA {
			if info.GitSHA == "unknown" || info.GitSHA == "" {
				status = "\033[33mNO INFO\033[0m"
			} else {
				status = "\033[31mSTALE\033[0m"
			}
		}

		fmt.Printf("%-20s %-10s %-10s %-22s %s\n",
			svc.name, info.Version, info.GitSHA, info.BuildTime, status)
	}
}
```

## Riesgos

| Riesgo | Mitigacion |
|--------|-----------|
| `--no-cache` es lento (rebuild from scratch) | `--cache` flag disponible para rebuilds rapidos cuando el cambio es solo Go code |
| Servicios no responden a `/v1/info` durante startup | Deploy script espera health check primero, luego verifica info |
| VERSION files no se actualizan | Este plan no resuelve version bumping automatico — es un fix del mecanismo de inyeccion, no del versionado |
| Extractor (Python) no tiene `/v1/info` | Fuera de scope. Se documenta como follow-up |
| YAML anchors `<<: *build-args` no soportados en todos los parsers | Docker Compose v2 lo soporta. Verificado por uso existente de `<<: *service-defaults` |
